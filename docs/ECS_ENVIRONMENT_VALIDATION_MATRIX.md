## Reproduce

```bash
python -m config.config_validation --all
python - <<'PY'
from config.environment_loader import get_environment_config
for e in ("local","dev","sit","uat","prod"):
    c = get_environment_config(env=e)
    assert len(c["applications"]) == 15 and "postgres" in c["databases"]
    print(e, "OK")
PY
```
